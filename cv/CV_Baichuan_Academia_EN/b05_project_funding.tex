\cventry{2024 -- 2027}{Toward a Resilient Transportation System, \$279,600}{\newline \underline{Principal Investigator (PI)}}{ \textit{National Science Foundation for Talented Young Researchers, China}}{}{
\begin{itemize}
\item {The project aims to develop a framework to improve the cities' capability of handling unplanned incidents and service disruptions, especially in public transit systems}
\item {The project focuses on monitoring, control and planning tasks that help the city to understand and mitigate incident impact, with metheds of machine learning and optimization}
\end{itemize}}
